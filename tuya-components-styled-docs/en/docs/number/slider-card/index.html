    Demo

    Studio

    import { ClassicSliderCard, ClassicLargeSliderCard } from 'tuya-panel-classic-kit';
    

    <ClassicSliderCard
      style={{ marginTop: 20 }}
      title="滑动card"
      icon={TuyaRNSvgs.power}
      value={value}
      unit="%"
      handValueChange={setValue}
      canTouchTrack
    />
    <ClassicLargeSliderCard
      icon={TuyaRNSvgs.power}
      showIconBg
      iconBgColor={{
        deg: 90,
        stops: {
          '0%': 'red',
          '100%': 'yellow',
        },
      }}
      style={{ marginTop: 20 }}
      title="粗的滑动条"
      unit="条"
      value={value}
      handSlidingComplete={setValue}
      disabled
    />
